Danh sách Parameters

Tham số Mô tả Ví dụ Mặc định
--ar Thay đổi tỷ lệ khung hình. --ar 16:9 4:3
--quality Thay đổi chất lượng của hình ảnh. --quality 1.5 1.0
--brightness Thay đổi độ sáng của hình ảnh. --brightness 0.2 0.0
--noise Thêm nhiễu vào hình ảnh. --noise 0.1 0.0
--seed Thiết lập hạt giống cho thuật toán tạo hình ảnh. --seed 1234 Ngẫu nhiên
--style Chọn phong cách nghệ thuật cho hình ảnh. --style Van Gogh None
--prompt Nhập lời nhắc cho Midjourney. --prompt Một con mèo đang chơi với quả bóng len màu đỏ trong vườn None
--iterations Số lượng vòng lặp để tạo hình ảnh. --iterations 200 100
--steps Số lượng bước trong mỗi vòng lặp. --steps 50 25
--width Chiều rộng của hình ảnh. --width 1024 512
--height Chiều cao của hình ảnh. --height 768 512
--upscale Tăng kích thước của hình ảnh sau khi tạo. --upscale 2 1
--denoising Mức độ khử nhiễu cho hình ảnh. --denoising 0.5 0.0
--cfg_scale Thay đổi độ mạnh của các tham số cấu hình. --cfg_scale 1.5 1.0
--chaos Thêm độ nhiễu và biến động vào hình ảnh. --chaos 0.5 0.0
--temperature Kiểm soát mức độ sáng tạo của hình ảnh. --temperature 1.2 1.0
--render_passes Số lượng lần render hình ảnh. --render_passes 2 1
--mask Sử dụng mặt nạ để che một phần hình ảnh. --mask mask.png None
--init_image Sử dụng hình ảnh khởi tạo để tạo ra hình ảnh mới. --init_image image.png None
--latent_diffusion Sử dụng kỹ thuật khuếch tán tiềm ẩn để tạo ra hình ảnh. --latent_diffusion 0.5 0.0
--superres Tăng kích thước của hình ảnh. --superres 2 1
--inpaint Khôi phục các phần bị thiếu trong hình ảnh. --inpaint inpaint.png None
--fast Tạo hình ảnh nhanh hơn (có thể làm giảm chất lượng). --fast False
--iw Chọn độ rộng mong muốn của hình ảnh (0 = tự động, 1 = nhỏ, 2 = lớn). --iw 1 0
--no Bỏ qua các tính năng nâng cao như phong cách, lặp lại, v.v. --no style False
--style random Chọn phong cách ngẫu nhiên. --style random None
--relax Làm cho hình ảnh ít chi tiết hơn. --relax False
--repeat Lặp lại lời nhắc nhiều lần để tạo ra các biến thể. --repeat 3 1
--stop Dừng tạo hình ảnh sau một số bước nhất định. --stop 50 100
--stylize Tăng cường phong cách của hình ảnh. --stylize False
--tile Tạo hình ảnh dạng ô. --tile False
--turbo Tăng tốc độ tạo hình ảnh (có thể làm giảm chất lượng). --turbo False
--weird Thêm độ kỳ ảo vào hình ảnh. --weird 1000 0
--niji Kích hoạt chế độ Niji để tạo ra hình ảnh theo phong cách anime. --niji False
--version Hiển thị phiên bản của Midjourney. --version None
